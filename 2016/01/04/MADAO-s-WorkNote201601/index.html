  • 学习了一些加密的知识，项目中用了MD5做签名，DES加密。MD5一般式不可逆的，项目中使用的时候对所传的参数包括Key也都进行了一次MD5算法的签名。 而DES加密则是利用服务器配置的公钥，对部分参数进行加密处理，服务器再进行解密。 不同平台的DES算法差异性也大，主要是IV向量不同。 iOS中系统的SDK加密: 
CCCryptorStatus cryptStatus = CCCrypt(kCCEncrypt, kCCAlgorithmDES,
                                          kCCOptionPKCS7Padding | kCCOptionECBMode,
                                          keyPtr, kCCBlockSizeDES,
                                          NULL,
                                          [data bytes], dataLength,
                                          buffer, bufferSize,
                                          &numBytesEncrypted);

  • 在自己想做一个接口测试App的时候，发现HTTPS请求需要通过证书的时候，Status Code一直为0。下面为在iOS中发起https请求获得证书的一些方法：

    1、开发时候有自己有服务端证书，可以打在App的Bundle里：在Targets中选择Build Phases，在Copy Bundle Resouces中选择你要加入Bundle中的文件。已基于AFNetwork的YTKNetwork框架为例则代码如下： 

YTKNetworkConfig *networkConfig = [YTKNetworkConfig sharedInstance];
  [networkConfig setBaseUrl:@"https://192.168.0.12:8443"];
  AFSecurityPolicy *securityPolicy = [AFSecurityPolicy policyWithPinningMode:AFSSLPinningModePublicKey];
  securityPolicy.allowInvalidCertificates = YES;
  securityPolicy.validatesDomainName = NO;
  networkConfig.securityPolicy = securityPolicy;

  • 今天折腾了一天的CoreData，主要问题集中在多表关联和数据的存储方面，目前不知道MR_saveWithBlock 和 MR_saveToPersistentStoreAndWait的区别，用下来发现后者要好用一点。完成了MADAOPost的基本功能。对于Core Data的掌握还需要加强。发现几个坑，对同一个数据操作的时候，要在同一个Context下，否则会报明显错误，所以CoreData的使用一定要注意Context。另外一点是表互连，用到Add或者Remove来关联的时候，需要写实现方法，例子如下： 
- (void)addRequest_argumentsObject:(Arguments *)value
{
    NSMutableSet *flowDetail = [self.request_arguments mutableCopy];
    [flowDetail addObject:value];
    self.request_arguments = [flowDetail copy];
}
- (void)removeRequest_argumentsObject:(Arguments *)value;
{
    NSMutableSet *flowDetail = [self.request_arguments mutableCopy];
    [flowDetail removeObject:value];
    self.request_arguments = [flowDetail copy];
}

关联的数据都是用的NSSet，如果要对数据进行有序的展示，可能还需要转换成NSOrderedSet或者普通数组进行处理。一个例子： 

/**排序*/
    NSSortDescriptor *sortDes = [[NSSortDescriptor alloc] initWithKey:@"argumentID" ascending:YES];
    NSArray *sortArray = [NSArray arrayWithObject:sortDes];
    NSArray *argumentArray = [self.request.request_arguments sortedArrayUsingDescriptors:sortArray];

查找的话使用了一下NSPredicate，发现效果不太好，不是很会用，需要加强。目前使用的MagicalRecord的查找函数：MR_findFirstByAttribute 等函数。

- (UIImageView *)arrowView
{
    if (!_arrowView) {
//        UIImage *image = [UIImage imageNamed:MJRefreshSrcName(@"arrow.png")] ?: [UIImage imageNamed:MJRefreshFrameworkSrcName(@"arrow.png")];
        UIImage *image = [UIImage imageNamed:@"refreshIndicator_img"];
        UIImageView *arrowView = [[UIImageView alloc] initWithImage:image];
        [self addSubview:_arrowView = arrowView];
    }
    return _arrowView;
}
- (UIActivityIndicatorView *)loadingView
{
    if (!_loadingView) {
        UIActivityIndicatorView *loadingView = [[UIActivityIndicatorView alloc] initWithActivityIndicatorStyle:UIActivityIndicatorViewStyleWhite];
        loadingView.hidesWhenStopped = YES;
        [self addSubview:_loadingView = loadingView];
    }
    return _loadingView;
}

#pragma makr - 重写父类的方法
- (void)placeSubviews
{
    [super placeSubviews];
    if (super.stateLabel.hidden) return;

    BOOL noConstrainsOnStatusLabel = super.stateLabel.constraints.count == 0;

    if (super.lastUpdatedTimeLabel.hidden) {
        // 状态
        if (noConstrainsOnStatusLabel) super.stateLabel.frame = self.bounds;
    } else {
        CGFloat stateLabelH = super.mj_h * 0.3;
        // 状态
        if (noConstrainsOnStatusLabel) {
            super.stateLabel.mj_x = 0;
            super.stateLabel.mj_y = 20; //由0改为20
            super.stateLabel.mj_w = super.mj_w;
            super.stateLabel.mj_h = stateLabelH;
        }

        // 更新时间
        if (super.lastUpdatedTimeLabel.constraints.count == 0) {
            super.lastUpdatedTimeLabel.mj_x = 0;
            super.lastUpdatedTimeLabel.mj_y = stateLabelH + 20;  
            //由stateLabelH改为stateLabelH + 20
            super.lastUpdatedTimeLabel.mj_w = super.mj_w;
            super.lastUpdatedTimeLabel.mj_h = super.mj_h - super.lastUpdatedTimeLabel.mj_y;
        }
        self.stateLabel.textColor = [UIColor whiteColor];
        self.lastUpdatedTimeLabel.textColor = [UIColor whiteColor];
        //这里修改自体颜色
    }
    // 箭头的中心点
    CGFloat arrowCenterX = super.mj_w * 0.5;
    if (!super.stateLabel.hidden) {
        arrowCenterX -= 100;
    }
    CGFloat arrowCenterY = super.mj_h * 0.3 + 20;  
    //系数原来为0.5，现改为0.3 + 20,实践而来
    CGPoint arrowCenter = CGPointMake(arrowCenterX, arrowCenterY);

    // 箭头
    if (self.arrowView.constraints.count == 0) {
        self.arrowView.mj_size = self.arrowView.image.size;
        self.arrowView.center = arrowCenter;
    }

    // 圈圈
    if (self.loadingView.constraints.count == 0) {
        self.loadingView.center = arrowCenter;
    }
}
//子类必须重写，否则刷新状态不会改变
- (void)setState:(MJRefreshState)state
{
    MJRefreshCheckState

    // 根据状态做事情
    if (state == MJRefreshStateIdle) {
        if (oldState == MJRefreshStateRefreshing) {
            self.arrowView.transform = CGAffineTransformIdentity;

            [UIView animateWithDuration:MJRefreshSlowAnimationDuration animations:^{
                self.loadingView.alpha = 0.0;
            } completion:^(BOOL finished) {
                // 如果执行完动画发现不是idle状态，就直接返回，进入其他状态
                if (self.state != MJRefreshStateIdle) return;

                self.loadingView.alpha = 1.0;
                [self.loadingView stopAnimating];
                self.arrowView.hidden = NO;
            }];
        } else {
            [self.loadingView stopAnimating];
            self.arrowView.hidden = NO;
            [UIView animateWithDuration:MJRefreshFastAnimationDuration animations:^{
                self.arrowView.transform = CGAffineTransformIdentity;
            }];
        }
    } else if (state == MJRefreshStatePulling) {
        [self.loadingView stopAnimating];
        self.arrowView.hidden = NO;
        [UIView animateWithDuration:MJRefreshFastAnimationDuration animations:^{
            self.arrowView.transform = CGAffineTransformMakeRotation(0.000001 - M_PI);
        }];
    } else if (state == MJRefreshStateRefreshing) {
        self.loadingView.alpha = 1.0; // 防止refreshing -> idle的动画完毕动作没有被执行
        [self.loadingView startAnimating];
        self.arrowView.hidden = YES;
    }
}

2016-01-29

*实现3D Touch Peek andPop的简单步骤(例子为在TableView中的使用)：
1、控制视图器遵守协议
2、一般如下代码设置delegate 

[self registerForPreviewingWithDelegate:self sourceView:self.view];

如果是针对cell： 

[self registerForPreviewingWithDelegate:self sourceView:cell];

3、实现两个协议方法,例子如下： 

#pragma mark - UIViewControllerPreviewingDelegate
- (UIViewController *)previewingContext:(id<UIViewControllerPreviewing>)previewingContext viewControllerForLocation:(CGPoint)location
{
    NSIndexPath *indexPath = [self.mainTableView indexPathForCell:(UITableViewCell* )[previewingContext sourceView]];
MessageDetailController *messageDetailVC = [[MessageDetailController alloc] init];
messageDetailVC.messageCell = [self.mainTableView cellForRowAtIndexPath:indexPath];
messageDetailVC.delegate = self;
messageDetailVC.title = @"消息详情";
messageDetailVC.preferredContentSize = CGSizeMake(0, 350);
CGRect rect = CGRectMake(10, 0, self.view.frame.size.width - 20,70);
previewingContext.sourceRect = rect;
NSLog(@"=====%@",previewingContext);
return messageDetailVC;}  
- (void)previewingContext:(id<UIViewControllerPreviewing>)previewingContext commitViewController:(UIViewController *)viewControllerToCommit
{
    [self showViewController:viewControllerToCommit sender:self];
}

4、增加预览按钮previewActionItems：
在需要预览的VC下实现如下方法： 

 - (NSArray<id<UIPreviewActionItem>> *)previewActionItems
{
    UIPreviewAction *action1 = [UIPreviewAction actionWithTitle:@"标题1" style:UIPreviewActionStyleDefault handler:^(UIPreviewAction * _Nonnull action, UIViewController * _Nonnull previewViewController) {
        NSLog(@"标题1");
    }];
    UIPreviewAction *action2 = [UIPreviewAction actionWithTitle:@"标题2" style:UIPreviewActionStyleDestructive handler:^(UIPreviewAction * _Nonnull action, UIViewController * _Nonnull previewViewController) {
        NSLog(@"标题2");
    }];
    UIPreviewAction *action3 = [UIPreviewAction actionWithTitle:@"标题3" style:UIPreviewActionStyleSelected handler:^(UIPreviewAction * _Nonnull action, UIViewController * _Nonnull previewViewController) {
        NSLog(@"标题3");
    }];
    NSArray * actions = @[action1,action2,action3];
    return actions;
}
